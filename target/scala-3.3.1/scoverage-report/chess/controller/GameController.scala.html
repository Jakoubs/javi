23 
24   def parse(input: String): Command =
25     val trimmed = input.trim.toLowerCase
26 
27     val simpleCommands = Map(
28       "flip" -> Command.Flip,
29       "undo" -> Command.Undo,
30       "resign" -> Command.Resign,
31       "draw" -> Command.OfferDraw,
32       "new" -> Command.NewGame,
33       "newgame" -> Command.NewGame,
34       "help" -> Command.Help,
35       "?" -> Command.Help,
36       "quit" -> Command.Quit,
37       "exit" -> Command.Quit,
38       "q" -> Command.Quit
39     )
40 
41     simpleCommands.get(trimmed).getOrElse {
42       if trimmed.startsWith("moves ") then
43         val posStr = trimmed.drop(6).trim
44         Pos.fromAlgebraic(posStr)
45           .map(Command.ShowMoves.apply)
46           .getOrElse(Command.Unknown(s"Invalid position: $posStr"))
47       else
48         parseMove(trimmed)
49           .getOrElse(Command.Unknown(s"Unknown command: '$input'. Type 'help' for help."))
50     }
51 
52   private def parseMove(s: String): Option[Command] =
53     if s.length < 4 then None
54     else
55       for
56         from  <- Pos.fromAlgebraic(s.take(2))
57         to    <- Pos.fromAlgebraic(s.drop(2).take(2))
58         promo = s.drop(4).headOption.flatMap(charToPromotion)
59         _     <- if s.drop(4).headOption.isDefined && promo.isEmpty then None else Some(())
60       yield Command.MakeMove(Move(from, to, promo))
61 
62   private def charToPromotion(c: Char): Option[PieceType] = c match
63     case 'q' => Some(PieceType.Queen)
64     case 'r' => Some(PieceType.Rook)
65     case 'b' => Some(PieceType.Bishop)
66     case 'n' => Some(PieceType.Knight)
67     case _   => None

120   // ── Command dispatch ───────────────────────────────────────────────────────
121 
122   def handleCommand(app: AppState, cmd: Command): AppState =
123     import Command.*
124     cmd match
125       case MakeMove(move) => handleMove(app, move)
126       case ShowMoves(pos) => handleShowMoves(app, pos)
127       case Flip           => app.copy(highlights = Set.empty, message = None, flipped = !app.flipped)
128       case Undo           => handleUndo(app)
129       case Resign         => handleResign(app)
130       case OfferDraw      => handleDraw(app)
131       case NewGame        => AppState.initial.copy(message = Some(TerminalView.info("New game started.")))
132       case Help           => app.copy(message = Some(TerminalView.helpText))
133       case Quit           => app.copy(running = false, message = None)
134       case Unknown(msg)   => app.copy(message = Some(TerminalView.error(msg)), highlights = Set.empty)

136   // ── Move handler ───────────────────────────────────────────────────────────
137 
138   private def handleMove(app: AppState, move: Move): AppState =
139     // Game already over?
140     app.status match
141       case GameStatus.Checkmate(_) | GameStatus.Stalemate | GameStatus.Draw(_) =>
142         return app.copy(message = Some(TerminalView.error("Game is over. Type 'new' to start again.")))
143       case _ => ()
144 
145     val legal = MoveGenerator.legalMoves(app.game)
146 
147     // Check legality
148     if !legal.exists(m => m.from == move.from && m.to == move.to && m.promotion == move.promotion) then
149       // Maybe pawn promotion needed?
150       if legal.exists(m => m.from == move.from && m.to == move.to && m.promotion.isDefined) && move.promotion.isEmpty then
151         return app.copy(message = Some(TerminalView.error("Pawn promotion required. Append q/r/b/n (e.g. e7e8q).")))
152       else
153         return app.copy(message = Some(TerminalView.error("Illegal move.")), highlights = Set.empty)
154 
155     // Apply move
156     val newGame   = GameRules.applyMove(app.game, move)
157     val newStatus = GameRules.computeStatus(newGame)
158 
159     val msg = newStatus match
160       case GameStatus.Checkmate(loser)  =>
161         val winner = if loser == Color.White then "Black" else "White"
162         Some(TerminalView.success(s"Checkmate! $winner wins!"))
163       case GameStatus.Stalemate        => Some(TerminalView.info("Stalemate — draw!"))
164       case GameStatus.Draw(reason)     => Some(TerminalView.info(s"Draw by $reason."))
165       case GameStatus.Check(_)         => Some(TerminalView.info("Check!"))
166       case GameStatus.Playing          => None
167 
168     app.copy(
169       game       = newGame,
170       status     = newStatus,
171       lastMove   = Some(move),
172       highlights = Set.empty,
173       message    = msg,
174       drawOffer  = false
175     )

177   // ── Show legal moves for a square ─────────────────────────────────────────
178 
179   private def handleShowMoves(app: AppState, pos: Pos): AppState =
180     app.game.board.get(pos) match
181       case None =>
182         app.copy(message = Some(TerminalView.error(s"No piece on ${pos.toAlgebraic}.")), highlights = Set.empty)
183       case Some(piece) if piece.color != app.game.activeColor =>
184         app.copy(message = Some(TerminalView.error("That is not your piece.")), highlights = Set.empty)
185       case Some(_) =>
186         val targets = MoveGenerator.legalMovesFrom(app.game, pos).map(_.to).toSet
187         if targets.isEmpty then
188           app.copy(message = Some(TerminalView.info("No legal moves for that piece.")), highlights = Set.empty)
189         else
190           app.copy(highlights = targets, message = None)

194   private def handleUndo(app: AppState): AppState =
195     app.game.history.lastOption match
196       case None =>
197         app.copy(message = Some(TerminalView.error("Nothing to undo.")))
198       case Some(prev) =>
199         val newStatus = GameRules.computeStatus(prev)
200         app.copy(
201           game       = prev,
202           status     = newStatus,
203           lastMove   = None,
204           highlights = Set.empty,
205           message    = Some(TerminalView.info("Last move undone."))
206         )

210   private def handleResign(app: AppState): AppState =
211     val loser  = app.game.activeColor
212     val winner = if loser == Color.White then "Black" else "White"
213     val resigned = app.game
214     val newStatus = GameStatus.Checkmate(loser)  // reuse checkmate status for resign
215     app.copy(
216       status  = newStatus,
217       message = Some(TerminalView.info(s"${loser} resigns. $winner wins!"))
218     )

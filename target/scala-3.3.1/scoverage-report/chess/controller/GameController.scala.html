37 object CommandParser:
38 
39   def parse(input: String): Command =
40     val trimmed = input.trim.toLowerCase
41 
42     trimmed match
43       case "flip"               => Command.Flip
44       case "undo"               => Command.Undo
45       case "resign"             => Command.Resign
46       case "draw"               => Command.OfferDraw
47       case "new" | "newgame"    => Command.NewGame
48       case "help" | "?"         => Command.Help
49       case "quit" | "exit" | "q"=> Command.Quit
50       case s if s.startsWith("moves ") =>
51         val posStr = s.drop(6).trim
52         Pos.fromAlgebraic(posStr) match
53           case Some(pos) => Command.ShowMoves(pos)
54           case None      => Command.Unknown(s"Invalid position: $posStr")
55       case s =>
56         parseMove(s) match
57           case Some(cmd) => cmd
58           case None      => Command.Unknown(s"Unknown command: '$input'. Type 'help' for help.")
59 
60   private def parseMove(s: String): Option[Command] =
61     // Formats: "e2e4"  or  "e7e8q"
62     val cleaned = s.trim.toLowerCase
63     if cleaned.length < 4 then return None
64 
65     val fromStr   = cleaned.take(2)
66     val restStr   = cleaned.drop(2)
67     val toStr     = restStr.take(2)
68     val promoChar = restStr.drop(2).headOption
69 
70     for
71       from  <- Pos.fromAlgebraic(fromStr)
72       to    <- Pos.fromAlgebraic(toStr)
73       promo = promoChar.flatMap(charToPromotion)
74       _     <- if promoChar.isDefined && promo.isEmpty then None else Some(())
75     yield Command.MakeMove(Move(from, to, promo))
76 
77   private def charToPromotion(c: Char): Option[PieceType] = c match
78     case 'q' => Some(PieceType.Queen)
79     case 'r' => Some(PieceType.Rook)
80     case 'b' => Some(PieceType.Bishop)
81     case 'n' => Some(PieceType.Knight)
82     case _   => None

110   def run(console: ConsoleIO = StdConsoleIO): Unit =
111     var app = AppState.initial

126     console.clear()

129       status     = app.status,

136   // ── Command dispatch ───────────────────────────────────────────────────────
137 
138   def handleCommand(app: AppState, cmd: Command): AppState =
139     import Command.*
140     cmd match
141       case MakeMove(move) => handleMove(app, move)
142       case ShowMoves(pos) => handleShowMoves(app, pos)
143       case Flip           => app.copy(highlights = Set.empty, message = None, flipped = !app.flipped)
144       case Undo           => handleUndo(app)
145       case Resign         => handleResign(app)
146       case OfferDraw      => handleDraw(app)
147       case NewGame        => AppState.initial.copy(message = Some(TerminalView.info("New game started.")))
148       case Help           => app.copy(message = Some(TerminalView.helpText))
149       case Quit           => app.copy(running = false, message = None)
150       case Unknown(msg)   => app.copy(message = Some(TerminalView.error(msg)), highlights = Set.empty)
151 
152   // ── Move handler ───────────────────────────────────────────────────────────
153 
154   private def handleMove(app: AppState, move: Move): AppState =
155     // Game already over?
156     app.status match
157       case GameStatus.Checkmate(_) | GameStatus.Stalemate | GameStatus.Draw(_) =>
158         return app.copy(message = Some(TerminalView.error("Game is over. Type 'new' to start again.")))
159       case _ => ()
160 
161     val legal = MoveGenerator.legalMoves(app.game)
162 
163     // Check legality
164     if !legal.exists(m => m.from == move.from && m.to == move.to && m.promotion == move.promotion) then
165       // Maybe pawn promotion needed?
166       if legal.exists(m => m.from == move.from && m.to == move.to && m.promotion.isDefined) && move.promotion.isEmpty then
167         return app.copy(message = Some(TerminalView.error("Pawn promotion required. Append q/r/b/n (e.g. e7e8q).")))
168       else
169         return app.copy(message = Some(TerminalView.error("Illegal move.")), highlights = Set.empty)
170 
171     // Apply move
172     val newGame   = GameRules.applyMove(app.game, move)
173     val newStatus = GameRules.computeStatus(newGame)
174 
175     val msg = newStatus match
176       case GameStatus.Checkmate(loser)  =>
177         val winner = if loser == Color.White then "Black" else "White"
178         Some(TerminalView.success(s"Checkmate! $winner wins!"))
179       case GameStatus.Stalemate        => Some(TerminalView.info("Stalemate — draw!"))
180       case GameStatus.Draw(reason)     => Some(TerminalView.info(s"Draw by $reason."))
181       case GameStatus.Check(_)         => Some(TerminalView.info("Check!"))
182       case GameStatus.Playing          => None
183 
184     app.copy(
185       game       = newGame,
186       status     = newStatus,
187       lastMove   = Some(move),
188       highlights = Set.empty,
189       message    = msg,
190       drawOffer  = false
191     )

193   // ── Show legal moves for a square ─────────────────────────────────────────
194 
195   private def handleShowMoves(app: AppState, pos: Pos): AppState =
196     app.game.board.get(pos) match
197       case None =>
198         app.copy(message = Some(TerminalView.error(s"No piece on ${pos.toAlgebraic}.")), highlights = Set.empty)
199       case Some(piece) if piece.color != app.game.activeColor =>
200         app.copy(message = Some(TerminalView.error("That is not your piece.")), highlights = Set.empty)
201       case Some(_) =>
202         val targets = MoveGenerator.legalMovesFrom(app.game, pos).map(_.to).toSet
203         if targets.isEmpty then
204           app.copy(message = Some(TerminalView.info("No legal moves for that piece.")), highlights = Set.empty)
205         else
206           app.copy(highlights = targets, message = None)

208   // ── Undo ───────────────────────────────────────────────────────────────────
209 
210   private def handleUndo(app: AppState): AppState =
211     app.game.history.lastOption match
212       case None =>
213         app.copy(message = Some(TerminalView.error("Nothing to undo.")))
214       case Some(prev) =>
215         val newStatus = GameRules.computeStatus(prev)
216         app.copy(
217           game       = prev,
218           status     = newStatus,
219           lastMove   = None,
220           highlights = Set.empty,
221           message    = Some(TerminalView.info("Last move undone."))
222         )
223 
224   // ── Resign ─────────────────────────────────────────────────────────────────
225 
226   private def handleResign(app: AppState): AppState =
227     val loser  = app.game.activeColor
228     val winner = if loser == Color.White then "Black" else "White"
229     val resigned = app.game
230     val newStatus = GameStatus.Checkmate(loser)  // reuse checkmate status for resign
231     app.copy(
232       status  = newStatus,
233       message = Some(TerminalView.info(s"${loser} resigns. $winner wins!"))
234     )
235 
236   // ── Draw offer ─────────────────────────────────────────────────────────────

238   private def handleDraw(app: AppState): AppState =
239     if app.drawOffer then
240       // Second player accepted
241       app.copy(
242         status    = GameStatus.Draw("agreement"),
243         drawOffer = false,
244         message   = Some(TerminalView.info("Draw accepted."))
245       )
246     else
247       app.copy(
248         drawOffer = true,
249         message   = Some(TerminalView.info(s"${app.game.activeColor} offers a draw. Type 'draw' to accept."))
250       )

43   private def pawnMoves(state: GameState, pos: Pos, color: Color): List[Move] =
44     val dir = if color == Color.White then 1 else -1
45     val startRow = if color == Color.White then 1 else 6
46     val promRow = if color == Color.White then 7 else 0
47     val board = state.board
48     val moves = scala.collection.mutable.ListBuffer.empty[Move]
49 
50     def addMoves(to: Pos): Unit =
51       if to.row == promRow then
52         moves += Move(pos, to, Some(PieceType.Queen))
53         moves += Move(pos, to, Some(PieceType.Rook))
54         moves += Move(pos, to, Some(PieceType.Bishop))
55         moves += Move(pos, to, Some(PieceType.Knight))
56       else
57         moves += Move(pos, to)
58 
59     val oneStep = pos + (0, dir)
60     if oneStep.isValid && board.isEmpty(oneStep) then
61       addMoves(oneStep)
62 
63     val twoStep = pos + (0, 2 * dir)

65       moves += Move(pos, twoStep)
66 
67     val leftTarget = pos + (-1, dir)
68     if leftTarget.isValid then
69       if board.isOccupiedBy(leftTarget, color.opposite) then addMoves(leftTarget)

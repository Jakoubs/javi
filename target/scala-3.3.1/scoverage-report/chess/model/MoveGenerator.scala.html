25   private def pseudoLegalMoves(state: GameState): List[Move] =
26     state.board.allPiecesOf(state.activeColor).flatMap { case (pos, piece) =>
27       piece.pieceType match
28         case PieceType.Pawn   => pawnMoves(state, pos, piece.color)
29         case PieceType.Knight => knightMoves(state, pos, piece.color)
30         case PieceType.Bishop => slidingMoves(state, pos, piece.color, bishopDirs)
31         case PieceType.Rook   => slidingMoves(state, pos, piece.color, rookDirs)
32         case PieceType.Queen  => slidingMoves(state, pos, piece.color, queenDirs)
33         case PieceType.King   => kingMoves(state, pos, piece.color)
34     }

44   private def pawnMoves(state: GameState, pos: Pos, color: Color): List[Move] =
45     val dir       = if color == Color.White then 1 else -1
46     val startRow  = if color == Color.White then 1 else 6
47     val promRow   = if color == Color.White then 7 else 0
48     val board     = state.board
49 
50     def withPromotion(to: Pos): List[Move] =
51       if to.row == promRow then
52         List(PieceType.Queen, PieceType.Rook, PieceType.Bishop, PieceType.Knight)
53           .map(pt => Move(pos, to, Some(pt)))
54       else
55         List(Move(pos, to))
56 
57     // single step
58     val oneStep = pos + (0, dir)
59     val single  = if oneStep.isValid && board.isEmpty(oneStep) then withPromotion(oneStep) else Nil
60 
61     // double step from starting row
62     val twoStep = pos + (0, 2 * dir)
63     val double  =
64       if pos.row == startRow && board.isEmpty(oneStep) && board.isEmpty(twoStep)
65       then List(Move(pos, twoStep))
66       else Nil
67 
68     // diagonal captures (including en passant)
69     val captures = List(-1, 1).flatMap { dc =>
70       val target = pos + (dc, dir)
71       if !target.isValid then Nil
72       else if board.isOccupiedBy(target, color.opposite) then withPromotion(target)
73       else if state.enPassantTarget.contains(target)      then List(Move(pos, target))
74       else Nil
75     }
76 
77     single ++ double ++ captures

79   private def knightMoves(state: GameState, pos: Pos, color: Color): List[Move] =
80     val deltas = List((2,1),(2,-1),(-2,1),(-2,-1),(1,2),(1,-2),(-1,2),(-1,-2))
81     deltas.flatMap { (dc, dr) =>
82       val to = pos + (dc, dr)
83       Option.when(to.isValid && !state.board.isOccupiedBy(to, color))(Move(pos, to))
84     }

86   private def slidingMoves(
87     state: GameState, pos: Pos, color: Color, dirs: List[(Int, Int)]
88   ): List[Move] =
89     dirs.flatMap { (dc, dr) =>
90       LazyList
91         .iterate(pos + (dc, dr))(p => p + (dc, dr))
92         .takeWhile(_.isValid)
93         .foldLeft((List.empty[Move], false)) { case ((acc, blocked), to) =>
94           if blocked then (acc, true)
95           else if state.board.isOccupiedBy(to, color) then (acc, true)
96           else if state.board.isOccupiedBy(to, color.opposite) then (acc :+ Move(pos, to), true)
97           else (acc :+ Move(pos, to), false)
98         }
99         ._1
100     }

102   private def kingMoves(state: GameState, pos: Pos, color: Color): List[Move] =
103     val normal = queenDirs.flatMap { (dc, dr) =>
104       val to = pos + (dc, dr)
105       Option.when(to.isValid && !state.board.isOccupiedBy(to, color))(Move(pos, to))
106     }
107     normal ++ castlingMoves(state, pos, color)

111   private def castlingMoves(state: GameState, kingPos: Pos, color: Color): List[Move] =
112     val cr    = state.castlingRights
113     val board = state.board
114     val row   = if color == Color.White then 0 else 7
115 
116     // king must not currently be in check
117     if isAttackedBy(board, kingPos, color.opposite) then return Nil
118 
119     def canCastle(
120       kingSide: Boolean,
121       hasRight: Boolean
122     ): Option[Move] =
123       if !hasRight then None
124       else
125         val (rookCol, emptyRange, transitCols) =
126           if kingSide then (7, 5 to 6, List(5, 6))
127           else             (0, 1 to 3, List(3, 4))
128 
129         val rookPos = Pos(rookCol, row)
130         val isRookPresent = board.get(rookPos).contains(Piece(color, PieceType.Rook))
131         val pathClear     = emptyRange.forall(c => board.isEmpty(Pos(c, row)))
132         val noTransitCheck = transitCols.forall { c =>
133           !isAttackedBy(board, Pos(c, row), color.opposite)
134         }
135         if isRookPresent && pathClear && noTransitCheck then
136           val toCol = if kingSide then 6 else 2
137           Some(Move(kingPos, Pos(toCol, row)))
138         else None
139 
140     List(
141       canCastle(kingSide = true,  hasRight = if color == Color.White then cr.whiteKingSide  else cr.blackKingSide),
142       canCastle(kingSide = false, hasRight = if color == Color.White then cr.whiteQueenSide else cr.blackQueenSide)
143     ).flatten

147   /** Is `pos` attacked by any piece of `attacker` on `board`? */
148   def isAttackedBy(board: Board, pos: Pos, attacker: Color): Boolean =
149     val defender = attacker.opposite
150 
151     // pawn attacks
152     val pawnDir = if attacker == Color.White then 1 else -1
153     val pawnAttack = List(-1, 1).exists { dc =>
154       board.get(pos + (dc, -pawnDir)).contains(Piece(attacker, PieceType.Pawn))
155     }
156     if pawnAttack then return true
157 
158     // knight attacks
159     val knightAttack = List((2,1),(2,-1),(-2,1),(-2,-1),(1,2),(1,-2),(-1,2),(-1,-2)).exists { (dc, dr) =>
160       board.get(pos + (dc, dr)).contains(Piece(attacker, PieceType.Knight))
161     }
162     if knightAttack then return true
163 
164     // king attacks
165     val kingAttack = queenDirs.exists { (dc, dr) =>
166       board.get(pos + (dc, dr)).contains(Piece(attacker, PieceType.King))
167     }
168     if kingAttack then return true
169 
170     // sliding attacks
171     def slidingAttack(dirs: List[(Int, Int)], types: Set[PieceType]): Boolean =
172       dirs.exists { (dc, dr) =>
173         LazyList
174           .iterate(pos + (dc, dr))(p => p + (dc, dr))
175           .takeWhile(_.isValid)
176           .foldLeft((false, false)) { case ((found, blocked), cur) =>
177             if blocked then (false, true)
178             else board.get(cur) match
179               case None                                      => (false, false)
180               case Some(Piece(c, pt)) if c == attacker && types.contains(pt) => (true, true)
181               case _                                         => (false, true)
182           }
183           ._1
184       }
185 
186     slidingAttack(bishopDirs, Set(PieceType.Bishop, PieceType.Queen)) ||
187     slidingAttack(rookDirs,   Set(PieceType.Rook,   PieceType.Queen))

191   /** Apply move to board only (no state update) and check if own king is attacked */
192   private def leavesKingInCheck(state: GameState, move: Move): Boolean =
193     val nextBoard = applyMoveToBoard(state, move)
194     nextBoard.findKing(state.activeColor) match
195       case None      => false
196       case Some(pos) => isAttackedBy(nextBoard, pos, state.activeColor.opposite)

198   /** Minimal board-only application used for check detection */
199   private def applyMoveToBoard(state: GameState, move: Move): Board =
200     val board = state.board
201     board.get(move.from) match
202       case None => board
203       case Some(piece) =>
204         // en passant capture
205         val boardAfterEP =
206           if piece.pieceType == PieceType.Pawn && state.enPassantTarget.contains(move.to) then
207             val dir = if piece.color == Color.White then -1 else 1
208             board.remove(move.to + (0, dir))
209           else board
210 
211         // castling: move rook
212         val boardAfterCastle =
213           if piece.pieceType == PieceType.King then
214             val dc = move.to.col - move.from.col
215             if math.abs(dc) == 2 then
216               val row     = move.from.row
217               val rookCol = if dc > 0 then 7 else 0
218               val newCol  = if dc > 0 then 5 else 3
219               boardAfterEP.movePiece(Pos(rookCol, row), Pos(newCol, row))
220             else boardAfterEP
221           else boardAfterEP
222 
223         // promotion
224         val finalPiece = move.promotion match
225           case Some(pt) => Piece(piece.color, pt)
226           case None     => piece
227 
228         boardAfterCastle.remove(move.from).put(move.to, finalPiece)

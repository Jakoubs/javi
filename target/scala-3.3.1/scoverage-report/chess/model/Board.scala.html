5 /** Immutable 8×8 board represented as a Map from Pos to Piece */
6 case class Board(pieces: Map[Pos, Piece]):
7 
8   def get(pos: Pos): Option[Piece] = pieces.get(pos)
9 
10   def isOccupied(pos: Pos): Boolean = pieces.contains(pos)
11 
12   def isOccupiedBy(pos: Pos, color: Color): Boolean =
13     pieces.get(pos).exists(_.color == color)
14 
15   def isEmpty(pos: Pos): Boolean = !pieces.contains(pos)
16 
17   /** Place a piece (overwrites) */
18   def put(pos: Pos, piece: Piece): Board = copy(pieces = pieces + (pos -> piece))
19 
20   /** Remove a piece */
21   def remove(pos: Pos): Board = copy(pieces = pieces - pos)
22 
23   /** Move without any special logic */
24   def movePiece(from: Pos, to: Pos): Board =
25     pieces.get(from) match
26       case None        => this
27       case Some(piece) => copy(pieces = (pieces - from) + (to -> piece))
28 
29   def findKing(color: Color): Option[Pos] =
30     pieces.collectFirst { case (pos, Piece(c, PieceType.King)) if c == color => pos }
31 
32   def allPiecesOf(color: Color): List[(Pos, Piece)] =
33     pieces.toList.filter(_._2.color == color)
34 
35   def toFenPlacement: String =
36     (7 to 0 by -1).map { row =>
37       val (parts, trailingEmpties) =
38         (0 to 7).foldLeft((List.empty[String], 0)) { case ((acc, emptyCount), col) =>
39           pieces.get(Pos(col, row)) match
40             case Some(piece) =>
41               val nextAcc = if emptyCount > 0 then acc :+ emptyCount.toString else acc
42               (nextAcc :+ piece.letter, 0)
43             case None =>
44               (acc, emptyCount + 1)
45         }
46       val completed = if trailingEmpties > 0 then parts :+ trailingEmpties.toString else parts
47       completed.mkString
48     }.mkString("/")

69   def fromFenPlacement(placement: String): Either[String, Board] =
70     val ranks = placement.split("/")
71     if ranks.length != 8 then Left("FEN placement must contain 8 ranks")
72     else
73       ranks.zipWithIndex.foldLeft[Either[String, Map[Pos, Piece]]](Right(Map.empty)) {
74         case (Left(error), _) => Left(error)
75         case (Right(acc), (rank, rankIndex)) =>
76           parseRank(rank, 7 - rankIndex).map(acc ++ _)
77       }.map(Board(_))

79   private def parseRank(rank: String, row: Int): Either[String, Map[Pos, Piece]] =
80     rank.foldLeft[Either[String, (Int, Map[Pos, Piece])]](Right((0, Map.empty))) {
81       case (Left(error), _) => Left(error)
82       case (Right((col, acc)), char) =>
83         if char.isDigit then
84           val emptySquares = char.asDigit
85           if emptySquares < 1 || emptySquares > 8 then Left(s"Invalid empty-square count '$char' in FEN")
86           else if col + emptySquares > 8 then Left(s"Rank '$rank' exceeds 8 files in FEN")
87           else Right((col + emptySquares, acc))
88         else
89           Piece.fromFenChar(char) match
90             case None => Left(s"Invalid piece '$char' in FEN")
91             case Some(piece) =>
92               if col >= 8 then Left(s"Rank '$rank' exceeds 8 files in FEN")
93               else Right((col + 1, acc + (Pos(col, row) -> piece)))
94     }.flatMap { (fileCount, pieces) =>
95       if fileCount == 8 then Right(pieces)
96       else Left(s"Rank '$rank' must describe exactly 8 files in FEN")
97     }

101 /**
102  * Complete, immutable game state.
103  *
104  * @param board          current board
105  * @param activeColor    whose turn it is

107  * @param enPassantTarget square a pawn may capture en-passant, if any

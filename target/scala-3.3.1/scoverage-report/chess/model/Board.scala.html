35   def toFenPlacement: String =
36     val ranks = scala.collection.mutable.ListBuffer.empty[String]
37     var row = 7
38     while row >= 0 do
39       val parts = scala.collection.mutable.ListBuffer.empty[String]
40       var emptyCount = 0
41       var col = 0
42       while col < 8 do
43         pieces.get(Pos(col, row)) match
44           case Some(piece) =>
45             if emptyCount > 0 then
46               parts += emptyCount.toString
47               emptyCount = 0
48             parts += piece.letter
49           case None =>
50             emptyCount += 1
51         col += 1
52       if emptyCount > 0 then parts += emptyCount.toString
53       ranks += parts.mkString
54       row -= 1
55     ranks.mkString("/")

82   def fromFenPlacement(placement: String): Either[String, Board] =
83     val ranks = placement.split("/")
84     if ranks.length != 8 then Left("FEN placement must contain 8 ranks")
85     else
86       var pieces = Map.empty[Pos, Piece]
87       var rankIndex = 0
88       while rankIndex < ranks.length do
89         parseRank(ranks(rankIndex), 7 - rankIndex) match
90           case Left(error) => return Left(error)
91           case Right(parsedRank) =>
92             pieces ++= parsedRank
93         rankIndex += 1
94       Right(Board(pieces))

96   private def parseRank(rank: String, row: Int): Either[String, Map[Pos, Piece]] =
97     var col = 0
98     var pieces = Map.empty[Pos, Piece]
99     var index = 0
100     while index < rank.length do
101       val char = rank(index)
102       if char.isDigit then
103         val emptySquares = char.asDigit
104         if emptySquares < 1 || emptySquares > 8 then
105           return Left(s"Invalid empty-square count '$char' in FEN")
106         if col + emptySquares > 8 then
107           return Left(s"Rank '$rank' exceeds 8 files in FEN")
108         col += emptySquares
109       else
110         Piece.fromFenChar(char) match
111           case None =>
112             return Left(s"Invalid piece '$char' in FEN")
113           case Some(piece) =>
114             if col >= 8 then
115               return Left(s"Rank '$rank' exceeds 8 files in FEN")
116             pieces += Pos(col, row) -> piece
117             col += 1
118       index += 1
119     if col == 8 then Right(pieces)
120     else Left(s"Rank '$rank' must describe exactly 8 files in FEN")

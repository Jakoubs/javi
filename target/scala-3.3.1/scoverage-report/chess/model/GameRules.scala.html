9   /**
10    * Apply `move` to `state` and return the new state.
11    * Assumes the move is legal (use MoveGenerator.legalMoves to verify first).
12    */
13   def applyMove(state: GameState, move: Move): GameState =
14     val board  = state.board
15     val piece  = board.get(move.from).get   // safe: called after legality check
16     val color  = piece.color
17 
18     // ── 1. Handle en-passant capture ──────────────────────────────────────
19     val boardAfterEP =
20       if piece.pieceType == PieceType.Pawn && state.enPassantTarget.contains(move.to) then
21         val capturedRow = if color == Color.White then move.to.row - 1 else move.to.row + 1
22         board.remove(Pos(move.to.col, capturedRow))
23       else board
24 
25     // ── 2. Handle castling (move rook) ────────────────────────────────────
26     val boardAfterCastle =
27       if piece.pieceType == PieceType.King then
28         val dc = move.to.col - move.from.col
29         if math.abs(dc) == 2 then
30           val row     = move.from.row
31           val rookCol = if dc > 0 then 7 else 0
32           val newCol  = if dc > 0 then 5 else 3
33           boardAfterEP.movePiece(Pos(rookCol, row), Pos(newCol, row))
34         else boardAfterEP
35       else boardAfterEP
36 
37     // ── 3. Move piece (with optional promotion) ───────────────────────────
38     val movedPiece = move.promotion match
39       case Some(pt) => Piece(color, pt)
40       case None     => piece
41 
42     val boardAfterMove =
43       boardAfterCastle.remove(move.from).put(move.to, movedPiece)
44 
45     // ── 4. Update castling rights ─────────────────────────────────────────
46     val cr = updateCastlingRights(state.castlingRights, piece, move)
47 
48     // ── 5. Update en-passant target ───────────────────────────────────────
49     val newEP =
50       if piece.pieceType == PieceType.Pawn && math.abs(move.to.row - move.from.row) == 2 then
51         val epRow = (move.from.row + move.to.row) / 2
52         Some(Pos(move.from.col, epRow))
53       else None
54 
55     // ── 6. Update clocks ──────────────────────────────────────────────────
56     val isCapture    = board.isOccupied(move.to) || state.enPassantTarget.contains(move.to)
57     val isPawnMove   = piece.pieceType == PieceType.Pawn
58     val newHalfClock = if isCapture || isPawnMove then 0 else state.halfMoveClock + 1
59     val newFullMove  = if color == Color.Black then state.fullMoveNumber + 1 else state.fullMoveNumber
60 
61     val newState = state.withHistory.copy(
62       board           = boardAfterMove,
63       activeColor     = color.opposite,
64       castlingRights  = cr,
65       enPassantTarget = newEP,
66       halfMoveClock   = newHalfClock,
67       fullMoveNumber  = newFullMove
68     )
69 
70     newState
71 
72   // ── Castling rights update ─────────────────────────────────────────────────
73 
74   private def updateCastlingRights(cr: CastlingRights, piece: Piece, move: Move): CastlingRights =
75     var updated = cr
76     // King moved → lose both sides
77     if piece.pieceType == PieceType.King then
78       updated = if piece.color == Color.White then updated.disableWhite else updated.disableBlack
79     // Rook moved from corner
80     if piece.pieceType == PieceType.Rook then
81       move.from match
82         case Pos(0, 0) => updated = updated.disableWhiteQueenSide
83         case Pos(7, 0) => updated = updated.disableWhiteKingSide
84         case Pos(0, 7) => updated = updated.disableBlackQueenSide
85         case Pos(7, 7) => updated = updated.disableBlackKingSide
86         case _         => ()
87     // Rook captured on corner
88     move.to match
89       case Pos(0, 0) => updated = updated.disableWhiteQueenSide
90       case Pos(7, 0) => updated = updated.disableWhiteKingSide
91       case Pos(0, 7) => updated = updated.disableBlackQueenSide
92       case Pos(7, 7) => updated = updated.disableBlackKingSide
93       case _         => ()
94     updated

98   def computeStatus(state: GameState): GameStatus =
99     val moves    = MoveGenerator.legalMoves(state)
100     val inCheck  = MoveGenerator.isInCheck(state, state.activeColor)
101 
102     if moves.isEmpty then
103       if inCheck then GameStatus.Checkmate(state.activeColor)
104       else            GameStatus.Stalemate
105     else if inCheck then
106       GameStatus.Check(state.activeColor)
107     else if isFiftyMoveRule(state) then
108       GameStatus.Draw("50-move rule")
109     else if isThreefoldRepetition(state) then
110       GameStatus.Draw("threefold repetition")
111     else if isInsufficientMaterial(state) then
112       GameStatus.Draw("insufficient material")
113     else
114       GameStatus.Playing

117 
118   private def isFiftyMoveRule(state: GameState): Boolean =
119     state.halfMoveClock >= 100
120 
121   private def isThreefoldRepetition(state: GameState): Boolean =
122     // Compare board + active color + castling rights + en-passant (FEN-like key)
123     val key = positionKey(state)
124     (state :: state.history).count(positionKey(_) == key) >= 3
125 
126   private def positionKey(state: GameState): String =
127     val pieces = state.board.pieces.toList.sortBy((p, _) => (p.col, p.row))
128       .map((p, pc) => s"${p.col}${p.row}${pc.letter}").mkString
129     s"$pieces|${state.activeColor}|${state.castlingRights}|${state.enPassantTarget}"
130 
131   private def isInsufficientMaterial(state: GameState): Boolean =
132     val pieces = state.board.pieces.values.toList
133     val types  = pieces.map(_.pieceType)
134     types match
135       case List(PieceType.King)                       => true  // KK (shouldn't happen but safe)
136       case t if t.count(_ == PieceType.King) == 2 &&
137                 t.size == 2                           => true  // K vs K
138       case t if t.count(_ == PieceType.King) == 2 &&
139                 t.size == 3 &&
140                 (t.contains(PieceType.Bishop) || t.contains(PieceType.Knight)) => true  // K+B vs K, K+N vs K
141       case _ => false

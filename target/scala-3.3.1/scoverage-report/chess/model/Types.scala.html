104   def disableBlack: CastlingRights = copy(blackKingSide = false, blackQueenSide = false)
105   def disableWhiteKingSide:  CastlingRights = copy(whiteKingSide  = false)
106   def disableWhiteQueenSide: CastlingRights = copy(whiteQueenSide = false)

110   def toFen: String =
111     val rights =
112       List(
113         Option.when(whiteKingSide)("K"),
114         Option.when(whiteQueenSide)("Q"),
115         Option.when(blackKingSide)("k"),
116         Option.when(blackQueenSide)("q")
117       ).flatten.mkString
118     if rights.nonEmpty then rights else "-"

120 object CastlingRights:

122     if value == "-" then Some(CastlingRights(false, false, false, false))
123     else if value.nonEmpty && value.forall("KQkq".contains(_)) && value.distinct.length == value.length then

57 case class Pos(col: Int, row: Int):
58   def isValid: Boolean = col >= 0 && col <= 7 && row >= 0 && row <= 7
59 
60   def +(dc: Int, dr: Int): Pos = Pos(col + dc, row + dr)

62   /** Algebraic notation, e.g. "e4" */
63   def toAlgebraic: String = s"${('a' + col).toChar}${row + 1}"

65 object Pos:
66   /** Parse "e4" → Pos(4, 3) */
67   def fromAlgebraic(s: String): Option[Pos] =
68     if s.length != 2 then None
69     else
70       val col = s(0) - 'a'
71       val row = s(1) - '1'

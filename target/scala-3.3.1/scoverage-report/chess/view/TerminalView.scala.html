31 
32   def render(
33               state:      GameState,
34               status:     GameStatus,
35               highlights: Set[Pos]   = Set.empty,   // legal-move targets
36               lastMove:   Option[Move] = None,
37               flipped:    Boolean      = false
38             ): String =
39     Seq(
40       header(state, status),
41       board(state, status, highlights, lastMove, flipped),
42       footer(state, status)
43     ).mkString("\n")

45   // ── Header (player names, captured pieces placeholder) ───────────────────
46 
47   private def header(state: GameState, status: GameStatus): String =
48     val (top, bottom) =
49       ("Black", "White")
50 
51     val checkStr = status match
52       case GameStatus.Check(c)     => s" ${FG_RED}${BOLD}CHECK!${RESET}"
53       case GameStatus.Checkmate(c) => s" ${FG_RED}${BOLD}CHECKMATE${RESET}"
54       case _                       => ""
55 
56     val moveStr = s"${DIM}Move ${state.fullMoveNumber}${RESET}"
57 
58     s"  $FG_CYAN$BOLD$top$RESET  $moveStr$checkStr\n"

60   // ── Board ─────────────────────────────────────────────────────────────────
61 
62   private def board(
63                      state:      GameState,
64                      status:     GameStatus,
65                      highlights: Set[Pos],
66                      lastMove:   Option[Move],
67                      flipped:    Boolean
68                    ): String =
69     val lastMovePosSet: Set[Pos] =
70       lastMove.map(m => Set(m.from, m.to)).getOrElse(Set.empty)
71 
72     val rows = if flipped then 0 to 7 else (7 to 0 by -1)
73     val cols = if flipped then (7 to 0 by -1).toList else (0 to 7).toList
74 
75     val boardRows = rows.map { row =>
76       val rankLabel = s"$FG_GRAY${BOLD} ${row + 1}  $RESET"
77       val cells = cols.map { col =>
78         val pos    = Pos(col, row)
79         val isLight = (col + row) % 2 == 1
80 
81         // Determine background
82         val bg =
83           if state.board.findKing(state.activeColor).contains(pos) &&
84             (status match { case GameStatus.Check(_) | GameStatus.Checkmate(_) => true; case _ => false })
85           then BG_RED
86           else if lastMovePosSet.contains(pos) then BG_YELLOW
87           else if highlights.contains(pos)     then BG_GREEN
88           else if isLight                      then BG_LIGHT
89           else                                      BG_DARK
90 
91         // Piece glyph width depends on the symbol we render:
92         // - Emoji/Unicode chess glyphs are often double-width (2 terminal columns).
93         // - Letters are single-width (1 terminal column).
94         // Every cell must occupy exactly 4 terminal columns:
95         //   piece (letter) : " A  "  = 1 space + letter(1) + 2 spaces = 4
96         //   empty           : "    " = 4 spaces                        = 4
97         //   dot             : " •  " = 1 + dot(1) + 2 spaces          = 4
98         state.board.get(pos) match
99           case None =>
100             if highlights.contains(pos) then s"$bg ${"•"}  $RESET"
101             else                             s"$bg    $RESET"
102           case Some(piece) =>
103             val fg = if piece.color == Color.White then FG_WHITE else s"$FG_BLACK$BOLD"
104             // Add one extra trailing space so that letters still align to 4 columns.
105             s"$bg$fg ${piece.letter}  $RESET"
106       }.mkString
107       
108       rankLabel + cells + "\n"
109     }.mkString
110 
111     // file labels: rank-label prefix is 4 chars wide, each cell is 4 columns
112     val fileLabels = cols.map { col =>
113       s"$FG_GRAY${BOLD}  ${('a' + col).toChar} $RESET"
114     }.mkString
115     
116     boardRows + "    " + fileLabels + "\n"
117 
118   // ── Footer (status message + prompt) ──────────────────────────────────────
119 
120   private def footer(state: GameState, status: GameStatus): String =
121     val turn = state.activeColor match
122       case Color.White => s"${FG_WHITE}${BOLD}White${RESET}"
123       case Color.Black => s"${FG_CYAN}${BOLD}Black${RESET}"
124 
125     val statusLine = status match
126       case GameStatus.Playing           => s"\n  $turn to move"
127       case GameStatus.Check(c)         =>
128         val col = if c == Color.White then s"${FG_WHITE}${BOLD}White${RESET}" else s"${FG_CYAN}${BOLD}Black${RESET}"
129         s"\n  $col is in check — $turn to move"
130       case GameStatus.Checkmate(loser) =>
131         val winner = if loser == Color.White then s"${FG_CYAN}${BOLD}Black${RESET}" else s"${FG_WHITE}${BOLD}White${RESET}"
132         s"\n  ${FG_RED}${BOLD}Checkmate!${RESET} $winner wins!"
133       case GameStatus.Stalemate        =>
134         s"\n  ${FG_YELLOW}${BOLD}Stalemate!${RESET} The game is a draw."
135       case GameStatus.Draw(reason)     =>
136         s"\n  ${FG_YELLOW}${BOLD}Draw${RESET} by $reason."
137 
138     statusLine + "\n"

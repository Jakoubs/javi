40   private def header(state: GameState, status: GameStatus): String =

42     val bottom = "White"

52   private def board(
53       state: GameState,
54       status: GameStatus,
55       highlights: Set[Pos],
56       lastMove: Option[Move],
57       flipped: Boolean
58   ): String =
59     val lastMovePosSet: Set[Pos] =
60       lastMove.map(m => Set(m.from, m.to)).getOrElse(Set.empty)
61 
62     val rows: Iterable[Int] = if flipped then 0 to 7 else (7 to 0 by -1)
63     val cols: List[Int] = if flipped then (7 to 0 by -1).toList else (0 to 7).toList
64 
65     val sb = new StringBuilder
66 
67     rows.foreach { row =>
68       sb.append(s"$FG_GRAY${BOLD} ${row + 1}  $RESET")
69 
70       cols.foreach { col =>
71         val pos = Pos(col, row)
72         val isLight = (col + row) % 2 == 1
73 
74         val inCheck =
75           state.board.findKing(state.activeColor).contains(pos) &&
76             (status match
77               case GameStatus.Check(_) | GameStatus.Checkmate(_) => true
78               case _ => false
79             )
80 
81         val bg =
82           if inCheck then BG_RED
83           else if lastMovePosSet.contains(pos) then BG_YELLOW
84           else if highlights.contains(pos) then BG_GREEN
85           else if isLight then BG_LIGHT
86           else BG_DARK
87 
88         val cell =
89           state.board.get(pos) match
90             case None =>
91               if highlights.contains(pos) then s"$bg ${"•"}  $RESET"
92               else s"$bg    $RESET"
93             case Some(piece) =>
94               val fg = if piece.color == Color.White then FG_WHITE else s"$FG_BLACK$BOLD"
95               s"$bg$fg ${piece.symbol} $RESET"
96 
97         sb.append(cell)
98       }
99 
100       sb.append("\n")
101     }
102 
103     // file labels: cells are 4 columns wide; prefix is 4 chars (" N  ")
104     sb.append("    ")
105     cols.foreach { col =>
106       sb.append(s"$FG_GRAY${BOLD}  ${('a' + col).toChar} $RESET")
107     }
108     sb.append("\n")
109     sb.toString

111   private def footer(state: GameState, status: GameStatus): String =
112     val turn = state.activeColor match
113       case Color.White => s"${FG_WHITE}${BOLD}White${RESET}"
114       case Color.Black => s"${FG_CYAN}${BOLD}Black${RESET}"
115 
116     val statusLine = status match
117       case GameStatus.Playing =>
118         s"\n  $turn to move"
119       case GameStatus.Check(c) =>
120         val col =
121           if c == Color.White then s"${FG_WHITE}${BOLD}White${RESET}"
122           else s"${FG_CYAN}${BOLD}Black${RESET}"
123         s"\n  $col is in check — $turn to move"
124       case GameStatus.Checkmate(loser) =>
125         val winner =
126           if loser == Color.White then s"${FG_CYAN}${BOLD}Black${RESET}"
127           else s"${FG_WHITE}${BOLD}White${RESET}"
128         s"\n  ${FG_RED}${BOLD}Checkmate!${RESET} $winner wins!"
129       case GameStatus.Stalemate =>
130         s"\n  ${FG_YELLOW}${BOLD}Stalemate!${RESET} The game is a draw. 🤝"
131       case GameStatus.Draw(reason) =>
132         s"\n  ${FG_YELLOW}${BOLD}Draw${RESET} by $reason. 🤝"
133 
134     statusLine + "\n"
